%%plot(ham);
y = s_a;
  x = 1:1:240;
  for i = 1:1:240
    x(i) = ham(i)*y(i+200);
  end

  per = 20;
  r = zeros(per, 1);
  r = r*0;
  r0 = 0.0;
  for j = 1:1:240
     r0 = r0 + x(i)*x(i);
  end
  for i = 1:1:per
    for j = per+1:1:240
        r(i) = r(i)+ x(j)*x(j-i);
      end
  end
